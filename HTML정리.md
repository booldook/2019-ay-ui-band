# HTML(HyperText Markup Language) 정리
~~~html
<!-- 기본태그 - block -->
<div>한 줄(block)을 차지</div>
<!-- 기본태그 - inline  -->
<a>글씨로 취급(inline)</a>
<img>
<span>
<b>

<!-- Semantic(의미론) tag - div의 속성을 가짐 /HTML5 -->
<header></header>
<section></section>
<footer></footer>

<!-- Headline 및 본문 -->
h1 ~ h5 : h1이 가장 크고 h5가 가장 작다./bold체로 표현
p : 본문의 문단을 표현
~~~
